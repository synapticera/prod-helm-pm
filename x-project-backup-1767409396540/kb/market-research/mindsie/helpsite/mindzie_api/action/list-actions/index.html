List Actions

Action

List and retrieve action details for your mindzieStudio project.

Overview

The Action Management endpoints allow you to discover and inspect actions configured for your project. Use these endpoints to list all available actions and retrieve detailed information about specific actions before execution.

List All Actions

GET /api/{tenantId}/{projectId}/action

Retrieve all actions configured for a specific project.

Request

GET https://your-mindzie-instance.com/api/{tenantId}/{projectId}/action
Authorization: Bearer {your-access-token}

Path Parameters

Parameter Type Required Description
tenantId GUID Yes Your tenant identifier
projectId GUID Yes Your project identifier

Response

Success (200 OK):

Response Fields

Field Type Description
actions array List of action objects
totalCount integer Total number of actions

Action Object Fields

Field Type Description
actionId GUID Unique identifier for the action
projectId GUID Project this action belongs to
name string Display name of the action
description string Description of what the action does
actionStatus string Current status (Idle, Running, etc.)
isEnabled boolean Whether the action is enabled
nextRunTime datetime Next scheduled execution time (if scheduled)
dateCreated datetime When the action was created
dateModified datetime When the action was last modified

Error Responses

Unauthorized (401):

HTTP/1.1 401 Unauthorized

{error message describing authorization failure}

Get Action Details

GET /api/{tenantId}/{projectId}/action/{actionId}

Retrieve detailed information about a specific action.

Request

GET https://your-mindzie-instance.com/api/{tenantId}/{projectId}/action/{actionId}
Authorization: Bearer {your-access-token}

Path Parameters

Parameter Type Required Description
tenantId GUID Yes Your tenant identifier
projectId GUID Yes Your project identifier
actionId GUID Yes The action to retrieve

Response

Success (200 OK):

{
  "actionId": "aaaaaaaa-bbbb-cccc-dddd-eeeeeeeeeeee",
  "projectId": "87654321-4321-4321-4321-210987654321",
  "name": "Daily Data Refresh",
  "description": "Refreshes data from source systems daily",
  "actionStatus": "Idle",
  "isEnabled": true,
  "nextRunTime": "2024-01-16T06:00:00Z",
  "lastRunTime": "2024-01-15T06:00:00Z",
  "configuration": "{\"sourceId\": \"...\", \"options\": {...}}",
  "dateCreated": "2024-01-01T10:00:00Z",
  "dateModified": "2024-01-15T14:30:00Z",
  "createdBy": "user@company.com",
  "modifiedBy": "user@company.com"
}

Response Fields

Field Type Description
actionId GUID Unique identifier for the action
projectId GUID Project this action belongs to
name string Display name of the action
description string Description of what the action does
actionStatus string Current status
isEnabled boolean Whether the action is enabled
nextRunTime datetime Next scheduled execution time
lastRunTime datetime Last execution time
configuration string JSON string containing action configuration
dateCreated datetime When the action was created
dateModified datetime When the action was last modified
createdBy string User who created the action
modifiedBy string User who last modified the action

Error Responses

Action Not Found (404):

{
  "error": "Action not found",
  "actionId": "aaaaaaaa-bbbb-cccc-dddd-eeeeeeeeeeee"
}

HTTP/1.1 401 Unauthorized

{error message describing authorization failure}

JavaScript/Node.js

const TENANT_ID = '12345678-1234-1234-1234-123456789012';
const PROJECT_ID = '87654321-4321-4321-4321-210987654321';
const BASE_URL = 'https://your-mindzie-instance.com';

// List all actions
const listActions = async (token) => {
  const url = `${BASE_URL}/api/${TENANT_ID}/${PROJECT_ID}/action`;

  const response = await fetch(url, {
    headers: {
      'Authorization': `Bearer ${token}`
    }
  });

  if (response.ok) {
    const result = await response.json();
    console.log(`Found ${result.totalCount} actions`);
    return result.actions;
  } else {
    throw new Error(`Failed to list actions: ${response.status}`);
  }
};

// Get action details
const getAction = async (actionId, token) => {
  const url = `${BASE_URL}/api/${TENANT_ID}/${PROJECT_ID}/action/${actionId}`;

  const response = await fetch(url, {
    headers: {
      'Authorization': `Bearer ${token}`
    }
  });

  if (response.ok) {
    return await response.json();
  } else if (response.status === 404) {
    throw new Error('Action not found');
  } else {
    throw new Error(`Failed to get action: ${response.status}`);
  }
};

// Example: List actions and get details for each
const inspectAllActions = async (token) => {
  const actions = await listActions(token);

  for (const action of actions) {
    console.log(`Action: ${action.name} (${action.actionId})`);
    console.log(`  Status: ${action.actionStatus}`);
    console.log(`  Enabled: ${action.isEnabled}`);
    console.log(`  Next Run: ${action.nextRunTime}`);
  }
};

Python

import requests

TENANT_ID = '12345678-1234-1234-1234-123456789012'
PROJECT_ID = '87654321-4321-4321-4321-210987654321'
BASE_URL = 'https://your-mindzie-instance.com'

def list_actions(token):
    """List all actions for the project."""
    url = f'{BASE_URL}/api/{TENANT_ID}/{PROJECT_ID}/action'
    headers = {
        'Authorization': f'Bearer {token}'
    }

    response = requests.get(url, headers=headers)

    if response.ok:
        result = response.json()
        print(f'Found {result["totalCount"]} actions')
        return result['actions']
    else:
        raise Exception(f'Failed to list actions: {response.status_code}')

def get_action(action_id, token):
    """Get details for a specific action."""
    url = f'{BASE_URL}/api/{TENANT_ID}/{PROJECT_ID}/action/{action_id}'
    headers = {
        'Authorization': f'Bearer {token}'
    }

    response = requests.get(url, headers=headers)

    if response.ok:
        return response.json()
    elif response.status_code == 404:
        raise Exception('Action not found')
    else:
        raise Exception(f'Failed to get action: {response.status_code}')

# Example: Find and execute enabled actions
def execute_enabled_actions(token):
    actions = list_actions(token)

    for action in actions:
        if action['isEnabled']:
            print(f'Action: {action["name"]}')
            print(f'  Status: {action["actionStatus"]}')
            print(f'  Next Run: {action["nextRunTime"]}')

            # Get full details
            details = get_action(action['actionId'], token)
            print(f'  Last Run: {details.get("lastRunTime", "Never")}')

C#

using System;
using System.Collections.Generic;
using System.Net.Http;
using System.Text.Json;
using System.Threading.Tasks;

public class ActionListResponse
{
    public List<ActionSummary> Actions { get; set; }
    public int TotalCount { get; set; }
}

public class ActionSummary
{
    public Guid ActionId { get; set; }
    public Guid ProjectId { get; set; }
    public string Name { get; set; }
    public string Description { get; set; }
    public string ActionStatus { get; set; }
    public bool IsEnabled { get; set; }
    public DateTime? NextRunTime { get; set; }
    public DateTime DateCreated { get; set; }
    public DateTime DateModified { get; set; }
}

public class ActionDetails : ActionSummary
{
    public DateTime? LastRunTime { get; set; }
    public string Configuration { get; set; }
    public string CreatedBy { get; set; }
    public string ModifiedBy { get; set; }
}

public class ActionApiClient
{
    private readonly HttpClient _httpClient;
    private readonly string _baseUrl;
    private readonly Guid _tenantId;
    private readonly Guid _projectId;

    public ActionApiClient(string baseUrl, Guid tenantId, Guid projectId, string accessToken)
    {
        _baseUrl = baseUrl;
        _tenantId = tenantId;
        _projectId = projectId;
        _httpClient = new HttpClient();
        _httpClient.DefaultRequestHeaders.Authorization =
            new System.Net.Http.Headers.AuthenticationHeaderValue("Bearer", accessToken);
    }

    public async Task<ActionListResponse> ListActionsAsync()
    {
        var url = $"{_baseUrl}/api/{_tenantId}/{_projectId}/action";
        var response = await _httpClient.GetAsync(url);

        if (response.IsSuccessStatusCode)
        {
            var json = await response.Content.ReadAsStringAsync();
            return JsonSerializer.Deserialize<ActionListResponse>(json,
                new JsonSerializerOptions { PropertyNameCaseInsensitive = true });
        }
        else
        {
            throw new Exception($"Failed to list actions: {response.StatusCode}");
        }
    }

    public async Task<ActionDetails> GetActionAsync(Guid actionId)
    {
        var url = $"{_baseUrl}/api/{_tenantId}/{_projectId}/action/{actionId}";
        var response = await _httpClient.GetAsync(url);

        if (response.IsSuccessStatusCode)
        {
            var json = await response.Content.ReadAsStringAsync();
            return JsonSerializer.Deserialize<ActionDetails>(json,
                new JsonSerializerOptions { PropertyNameCaseInsensitive = true });
        }
        else if (response.StatusCode == System.Net.HttpStatusCode.NotFound)
        {
            throw new Exception($"Action {actionId} not found");
        }
        else
        {
            throw new Exception($"Failed to get action: {response.StatusCode}");
        }
    }
}

Best Practices

  • Cache Action List: Cache the action list to avoid repeated API calls, refresh periodically
  • Check Enabled Status: Always check isEnabled before attempting to execute an action
  • Monitor Status: Use actionStatus to understand if an action is currently running
  • Parse Configuration: The configuration field contains JSON - parse it to understand action settings
  • Track Scheduling: Use nextRunTime to understand when scheduled actions will run next
